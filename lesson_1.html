var panel = Ext.create('Ext.panel.Panel', {
	title: 'ViewModel操作',
	viewModel: { // 定义
		data: {
			title: 'Hello World',
		}
	},
	//height:100,
	bodyStyle:'padding:8px 0 12px 8px;',
	bind:{
		html:'{title}'
	},
	items:[{
		xtype:'textfield',
		fieldLabel:'测试内容',
		bind: '{title}' // 取值 和 设值 // data 中的 title 属性
	}],
	buttons:[{
		text:'bind 操作',
		handler:function(btn){
			panel.getViewModel().set('title',btn.text); // 设值 //  data 中的 title 属性
		}
	}],
   // renderTo: Ext.getBody() /
	renderTo: 'displaye'
});
