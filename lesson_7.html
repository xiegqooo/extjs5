Ext.define('MyApp.controller.Main', {
    extend : 'Ext.app.Controller',
	// 地址路由
    routes : {
        'home:id' :{
			action     : 'onHome',
			//before     : 'beforeShowTab',
			conditions : {
				':id'    : '([0-9]+)', // id 必须是数字
			}
		}
		
    },
	// 事件监听
	control : {
		'#testRouter' : {
			click : 'onClickSubmit'
		}
	},
	// 显示结果
    onHome : function(id) {
		console.log('#home后的id值:'+id);
	},
	// 触发事件
	onClickSubmit:function(btn, e, eOpts){
		this.redirectTo('home1234');
	}
});
